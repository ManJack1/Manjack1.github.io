#+BEGIN_SRC emacs-lisp :exports code
(defun my-compile-and-run-c ()
  "编译并运行当前 C 文件"
  (interactive)
  (save-buffer)
  (let* ((file (buffer-file-name))
         (exe (file-name-sans-extension file)))
    (compile (format "gcc %s -o %s && %s" file exe exe))))

(global-set-key (kbd "C-c c") 'my-compile-and-run-c)
(global-set-key (kbd "C-c v") 'org-download-clipboard)
